Below is the code that you can add to your head html tag so that perf-trace-server will collect statistics from your webapp! 
        <script type = "text/javascript" >

            (function(window) {{ '{' }}
              // Setting your personal webappId
              window.ptParams = {{ '{' }}webappId: '1'{{ '}' }};

              // Sourcing the perf-trace script
              let scriptStr = 'script';
              let server = (document.location.protocol === 'https:'
                  ? 'https://github.com/andziaania/perf-trace-server/raw/master/src/main/resources/static/'
                  : 'http://pawelczyk.biz.pl:8081/');

              let pt = document.createElement(scriptStr);
              pt.type = 'text/javascript';
              pt.async = true;
              pt.src = server + 'perftrace.js';

              let firstScriptOnPage = document.getElementsByTagName('script')[0];
              firstScriptOnPage.parentNode.insertBefore(pt, firstScriptOnPage);

              {{ '}' }})(window);

        </script>
